import processing.video.*;
import jp.nyatla.nyar4psg.*;

Capture cam;
MultiMarker nya;

void setup() {
size(640,480,P3D);
colorMode(RGB, 100);
println(MultiMarker.VERSION);
cam=new Capture(this,640,480);
nya=new MultiMarker(this,width,height,"../../data/camera_para.dat",NyAR4PsgConfig.CONFIG_PSG);
nya.addARMarker("../../data/patt.hiro",80);
cam.start();
}

void draw(){
if (cam.available() !=true) {
return;
}
cam.read();
nya.detect(cam);
background(10);
nya.drawBackground(cam);
if((!nya.isExist(0))){
return;
}
nya.beginTransform(0);
fill(0,0,255);
translate(0,0,20);
triangle(100, 175, 168, 50, 86, 75);
nya.endTransform();

nya.beginTransform(0);
fill(255,161,231);
translate(0,0,20);
box(80, 75, 185);
nya.endTransform();


nya.beginTransform(0);
fill(0,0,255);
translate(0,0,20);
ellipse(20,30,90,40);
nya.endTransform();

nya.beginTransform(0);
fill(150);
strokeWeight(5);
translate(0,0,20);
ellipse(50,60,120,70);
nya.endTransform();
}
